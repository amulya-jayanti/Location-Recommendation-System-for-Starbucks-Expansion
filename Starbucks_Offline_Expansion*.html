def create_buffer(df,radius):
    '''
    Create a buffer around the geometry of a location
    '''
    if not df.crs:
        print('CRS not set, setting 4326')
        df.set_crs(epsg=4326, inplace=True)
    elif df.crs.to_epsg() == 4326:
        try:
            df = df.to_crs(epsg=3857) # Convert into 3857
            df['geometry'] = df['geometry'].buffer(radius) # Create a buffer in metres
            df = df.to_crs(epsg=4326) # Converting back to 4326
            return df
        except Exception as e:
            print(e)


def create_features(radius):
    '''
    Create features around a Starbucks location
    '''
    starbucks_buffer = create_buffer(starbucks,radius) # Create buffer

    st_buffer = gpd.sjoin(starbucks_buffer,all_pois).reset_index(drop=True) # Merge dataframes

    st_buffer_counts = st_buffer.groupby(['place_id_left','category'])['place_id_right'].count().reset_index() # Get the counts of brands/POI/hotel categoris for every starbucks

    st_buffer_counts = st_buffer_counts.pivot_table(index='place_id_left',columns='category',values='place_id_right').reset_index()

    st_buffer_counts.fillna(0,inplace=True)
    st_buffer_counts.columns = ['place_id'] + ['number_of_'+x+'_within_'+str(radius)+'m' for x in st_buffer_counts.columns if x!='place_id_left']

    return st_buffer_counts

def create_features_isochrone(df,min=5,contour='driving'):
    '''
    Create features around a Starbucks location
    '''
    st_buffer = gpd.sjoin(df,all_pois).reset_index(drop=True)
    st_buffer_counts = st_buffer.groupby(['place_id_left','category'])['place_id_right'].count().reset_index()
    st_buffer_counts = st_buffer_counts.pivot_table(index='place_id_left',columns='category',values='place_id_right').reset_index()
    st_buffer_counts.fillna(0,inplace=True)
    st_buffer_counts.columns = ['place_id'] + ['number_of_'+x+'_within_'+str(min)+'min_'+contour for x in st_buffer_counts.columns if x!='place_id_left']

    return st_buffer_counts

from haversine import haversine, Unit

def second_min_haversine_distance(row, B):
    '''
    The function returns the second smallest haversine distance between Starbucks locations. Second smallest, because the smallest distance would always be 0, which is the distance of a Starbucks to itself
    '''
    point_A = (row['latitude'], row['longitude'])

    # Compute distances along with corresponding B IDs
    distances = sorted(
        [(haversine(point_A, (b_lat, b_lng), unit=Unit.KILOMETERS), b_id)
         for b_id, b_lat, b_lng in zip(B['place_id'], B['latitude'], B['longitude'])]
    )

    non_zero_distances = [(d, b_id) for d, b_id in distances if d > 0]

    # If at least 2 points exist, return second closest distance & ID
    if len(non_zero_distances) > 1:
        return non_zero_distances[1]  # (distance, id)
    else:
        return (np.nan, np.nan)  # Return NaN if there's only one point

def create_buffer(df,radius):
    '''
    Create a buffer around the geometry of a location
    '''
    if not df.crs:
        print('CRS not set, setting 4326')
        df.set_crs(epsg=4326, inplace=True)
    elif df.crs.to_epsg() == 4326:
        try:
            df = df.to_crs(epsg=3857) # Convert into 3857
            df['geometry'] = df['geometry'].buffer(radius) # Create a buffer in metres
            df = df.to_crs(epsg=4326) # Converting back to 4326
            return df
        except Exception as e:
            print(e)
